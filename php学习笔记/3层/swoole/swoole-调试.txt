使用phpize生成php编译配置。 ./configure 做编译配置检测，make进行编译，make install安装

下载最新release版 编译安装

tar zxvf ./swoole.tar.gz && \
mv swoole-src* swoole-src && \
cd swoole-src && \
phpize && \
./configure \
--enable-openssl  \
--enable-http2  \
--enable-async-redis \
--enable-sockets \
--enable-mysqlnd  \
--enable-debug-log \
--enable-trace-log && \
make clean && make && sudo make install
